Create an inordered threaded binary search tree. Perform inorder, preorder traversals without recursion and deletion of a node. Analyze time and space complexity of the algorithm. 

                    
include <iostream>
using namespace std;


class node{
public:
    int data;
    node *left,*right;
    bool lth,rth;


    node(){
        data=0;
        left=NULL;
        right=NULL;
        lth=true;
        rth=true;
    }

    node(int data){
        this->data=data;
        left=NULL;
        right=NULL;
        lth=true;
        rth=true;

    }
};

class Tbst{
public:
    node*root;

    Tbst(){
        root=NULL;
    }

    void Insertnode(int key){
        node *cur,*par;
        par=NULL;
        cur=root;

            while(cur){             //seraching the node
                par=cur;

                if(cur->data==key){
                    cout<<"Node Already Exist."<<endl;    //Duplicate case handeled .
                    return ;
                }
                else if(key<cur->data){
                    if(cur->lth==0){
                        cur=cur->left;
                    }
                    else break;

                }

                else{
                    if(cur->rth==0){
                        cur=cur->right;
                    }
                    else break;
                }

            }

            //now we found out the place for insertion of the new node;

             node *temp=new node(key);

             //case 2 left child:
             if(par==NULL){
                 root=temp;
             }
             else if(key<par->data){
                 temp->left=par->left;
                 temp->right=par;

                 par->left=temp;
                 par->lth=false;

             }
             else if(key>par->data){
                 temp->right=par->right;
                 temp->left=par;

                 par->rth=false;
                 par->right=temp;
             }

    }

    void Create(){
        int n;

     node *inSuccessor(node *ptr)
        {
            if (ptr->rth == 1)
            {
                return ptr->right;
            }
            else
            {
                ptr = ptr->right;
                while (ptr->lth == 0)
                {
                    ptr = ptr->left;
                }
                return ptr;
            }
        }
        node *inPredecessor(node *ptr)
        {
            if (ptr->lth == 1)
            {
                return ptr->left;
            }
            ptr = ptr->left;

            while (ptr->rth == 0)
            {
                ptr = ptr->right;
            }

            return ptr;
        }


    void Inorder(node *r){
        if(!r){
            return ;
        }
        node *temp = r;

        while (temp->lth== 0)
        {
            temp = temp->left;
        }
        while (temp != NULL)
        {	cout << temp->data << " ";
            temp = inSuccessor(temp);
        }
    }

    void PreOrder(node *r){

        if(!r){
            return ;
        }
         while (r != NULL)
                    {
                        cout << r->data << " ";
                        if (r->lth == 0)
                        {
                            r = r->left;
                        }
                        else if (r->rth == 0)
                        {
                            r = r->right;
                        }
                        else
                        {
                            while (r != NULL && r->rth == 1)
                            {
                                r = inSuccessor(r);
                            }
                            if (r != NULL)
                            {
                                r = r->right;
                            }
                        }
                    }

    node* zeroChild(node *ro, node *par,
                    node *cur)
        {
            if (par == NULL)
                ro = NULL;

            else if (cur == par->left)
            {
                par->lth = 1;
                par->left = cur->left;
            }
            else
            {
                par->rth = 1;
                par->right = cur->right;
            }
            delete(cur);
            return ro;
        }

        node *oneChild(node *ro, node *par,
                    node *ptr)
        {
            node *child;

            if (ptr->lth == 0)
                child = ptr->left;
            else
                child = ptr->right;

            if (par == NULL)
                ro = child;
            else if (ptr == par->left)
                par->left = child;
            else
                par->right = child;

            node *s = inSuccessor(ptr);
            node *p = inPredecessor(ptr);

            if (ptr->lth == 0)
                p->right = s;

            else
            {
                if (ptr->rth == 0)
                    s->left = p;
            }

            delete(ptr);
            return ro;
        }

        node *twoChild(node *ro, node *par,
                    node *ptr)
        {

            node *parentsucc = ptr;
            node *succ = ptr->right;
            while (succ->lth == 0)
            {
                parentsucc = succ;
                succ = succ->left;
            }

            ptr->data = succ->data;

            if (succ->lth== 1 && succ->rth == 1)
                ro = zeroChild(ro, parentsucc, succ);
            else
                ro = oneChild(ro, parentsucc, succ);

            return ro;
        }

        node *deleteElement(node *ro, int val)
        {
            node *par = NULL, *ptr = ro;
            int flag = 0;
            while (ptr != NULL)
            {
                if (val == ptr->data)
                {
                    flag = 1;
                    break;
                }
                par = ptr;
                if (val < ptr->data)
                {
                    if (ptr->lth == 0)
                        ptr = ptr->left;
                    else
                        break;
                }
                else
                {
                    if (ptr->rth == 0)
                        ptr = ptr->right;
                    else
                        break;
                }
            }

            if (flag == 0){
                cout<<"The key to be deleted is not present in the tree!"<<endl;
            }
            else if (ptr->lth == 0 && ptr->rth == 0)
                ro = twoChild(ro, par, ptr);
            else if (ptr->lth == 0)
                ro = oneChild(ro, par, ptr);
            else if (ptr->rth == 0)
                ro = oneChild(ro, par, ptr);
            else
                ro = zeroChild(ro, par, ptr);

            return ro;
        }
