Problem statement

Consider telephone book database of N clients. Make use of a hash table implementation to quickly look up clientâ€˜s telephone number. Make use of two collision handling techniques and compare them using number of comparisons required to find a set of telephone numbers (Note: Use linear probing with replacement and without replacement) 

                    
#include <bits/stdc++.h>
using namespace std;

class node
{

public:
    string name;
    long long int mb;

    node()
    {
        name = "--";
        mb = 0;
    }

    friend class hashmap;
};

class hashmap
{

    int size;
    node *arr;

public:
    hashmap(int size)
    {
        this->size = size;
        arr = new node[size];
    }

    int hash(string name)
    {
        int sum = 0;
        for (int i = 0; i < name.size(); i++)
        {
            sum = (sum + (int(name[i])));
        }
        return sum % size;
    }

    void insertWithoutReplacement(string n, int mbl)
    {

        int index = hash(n);
        int cnt = 0;

        if (arr[index].name == "--" && arr[index].mb == 0)
        {
            arr[index].name = n;
            arr[index].mb = mbl;
            cnt++;
        }
        else
        {
            index++;
            while (arr[index].name != "--" && arr[index].mb != 0)
            {
                index++;
                index %= size;
            }
            arr[index].name = n;
            arr[index].mb = mbl;
            cnt++;
        }

        if (cnt == size)
        {
            cout << "Hash Table is full..." << endl;
        }
    }

    void insertWithReplacement(string n, int mbl)
    {

        int index = hash(n);
        int cnt = 0;

        if (arr[index].name == "--" && arr[index].mb == 0)
        {
            arr[index].name = n;
            arr[index].mb = mbl;
            cnt++;
        }
        else
        {
            string temp = arr[index].name;
            int tem = hash(temp);

            if (tem == index)
            {
                insertWithoutReplacement(n, mbl);
            }
            else
            {
                string copy = arr[index].name;
                int cop = arr[index].mb;
                arr[index].name = n;
                arr[index].mb = mbl;
                insertWithoutReplacement(copy, cop);
            }
        }
    }

    void searching(string n)
    {
        int index = hash(n);
        int iteration = 0;
        if (arr[index].name == n)
        {
            iteration++;
            cout << "Data is found --" << endl;
            cout << arr[index].name << " : " << arr[index].mb << endl;
        }
        else
        {
            int j = 1;
            bool t = false;
            while (j < size)
            {
                if (arr[(index + j) % size].name == n)
                {
                    iteration++;
                    cout << "Data is found --" << endl;
                    cout << arr[(index + j) % size].name << " : " << arr[(index + j) % size].mb << endl;
                    t = true;
                    break;
                }
                else
                {
                    iteration++;
                    j++;
                }
            }
            if (t == false)
            {
                cout << "NOt Present ..." << endl;
            }
        }

        cout << endl;
        cout << "Number of comparisions : " << iteration << endl;
    }
    void deleting(string n)
    {
        int index = hash(n);
        int iteration = 0;
        if (arr[index].name == n)
        {
            cout << "Data is Deleted Successfully --" << endl;
            arr[index].name = "--";
            arr[index].mb = 0;
            iteration++;
        }
        else
        {
            int j = 1;
            bool t = false;
            while (j < size)
            {
                if (arr[(index + j) % size].name == n)
                {
                    cout << "Data is Deleted Successfully --" << endl;
                    arr[index].name = "--";
                    arr[index].mb = 0;
                    iteration++;
                    t = true;
                    break;
                }
                else
                {
                    iteration++;
                    j++;
                }
            }
            if (t == false)
            {
                cout << "Data is NOt Present ..." << endl;
            }
        }
        cout << endl;
        cout << "Number of comparisions : " << iteration << endl;
    }

    void print()
    {
        for (int i = 0; i < size; i++)
        {
            cout << arr[i].name << " : " << arr[i].mb << endl;
        }
        cout << endl;
    }
};
