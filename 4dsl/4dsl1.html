Problem statement

Beginning with an empty binary search tree. Construct the binary search tree by inserting the values in given order. After constructing binary search tree perform following operations 1) Insert a new node 2) Find numbers of node in longest path 3) Minimum and maximum data value found in tree 4) Change a tree so that the roles of the left and right pointers are swapped at every node 5)Search an element 

                    

#include <iostream>
using namespace std;

class node    //Class Node For node of the tree
{
public:
    int data;
    node *left, *right;
    node()
    {
        left = NULL;
        right = NULL;
    }
    node(int x)
    {
        left = NULL;
        right = NULL;
        data = x;
    }
};
class Bst     //tree class for bst having root as data member
{
public:
    node *root;

    Bst()
    {
        root = NULL;
    }

    void create(int num)
    {                               //Normal Creation function
        int counter = 0;
        if (root == NULL)
        {
            node *r = new node;
            r->data = num;
            root = r;
        }
        else
        {
            node *p = root;
            bool flag = true;
            while (flag)
            {
                if (p->left != NULL && p->data > num)
                {
                    p = p->left;
                    counter++;
                }
                else if (p->left == NULL && p->data > num)
                {
                    node *n = new node;
                    n->data = num;
                    p->left = n;
                    p = p->left;
                    flag = false;
                    counter++;
                    break;
                }
                if (p->right != NULL && p->data < num)
                {
                    p = p->right;
                    counter++;
                }
                else if (p->right == NULL && p->data < num)
                {
                    node *n = new node;
                    n->data = num;
                    p->right = n;
                    p = p->right;
                    flag = false;
                    counter++;

                    break;
                }
            }
        }
    }
                       //Traversals
    void Preorder(node *p)
    {
        if (p)
        {
            cout << p->data << " ";  //Preorder   node-left-right
            Preorder(p->left);
            Preorder(p->right);
        }
        else
            return;
    }
    void Inorder(node *p)
    {
        if (p)
        {
            Inorder(p->left);
            cout << p->data << " ";   //Inorder   left-node-right
            Inorder(p->right);
        }
        else
            return;
    }
    void Postorder(node *p)
    {
        if (p)
        {
            Postorder(p->left);
            Postorder(p->right);              //postorder   left-right-node
            cout << p->data << " ";
        }
        else
            return;
    }

    int Height(node *p)
    {
        int x, y;
        if (p != NULL)                     //for calculating the height of the Tree .
        {
            x = Height(p->left);
            y = Height(p->right);

            return max(x + 1, y + 1);
        }

        return 0;
    }

   int Max(node *p)
    {
        while (p->right != NULL)
        {
            p = p->right;                          //max value is the rightmost node of the tree.
        }
        cout << "The Maximum Value in the tree is:- " << p->data << endl;
        return p->data;
    }
    int Min(node *p)
    {
        while (p->left != NULL)
        {                                         //min valu is the leftmost node of the tree
            p = p->left;
        }
        cout << "The Minimum Value in the tree is:- " << p->data << endl;
        return p->data;
    }

    node *Searchnode(node *&curr, node *&par, int Key)
    {
        //Searching the position of the new value,

        while (curr != NULL)
        {
            if (curr->data == Key)
            {
                return NULL;
            }
            else
            {
                par = curr;

                if (curr->data > Key)
                {
                    curr = curr->left;
                }
                else
                {
                    curr = curr->right;
                }
            }
        }
    }

     void Search(int key){
        node *curr = this->root;
        node *par = NULL;

        Searchnode(curr,par,key);
        if(curr){
            cout<<"The Node is found :- "<<curr->data<<endl;;
            return;
        }
        else{
            cout<<"The Node is Not Present in the Tree."<<endl;
            return;
        }
     }



    void insert(int x)              //Insert Function using Searching Technique.
    {
        node *curr = this->root;
        node *par = NULL;

        if (root == NULL)
        {
            node *t = new node(x);
            root = t;
        }
        else
        {
            Searchnode(curr, par, x);

            if (curr != NULL)
            {
                cout << "You Enterd a Node which is already Present" << endl;
                return;
            }
            else
            {
                if (par->data > x)
                {
                    node *t = new node(x);
                    par->left = t;
                    return;
                }
                else
                {
                    node *t = new node(x);
                    par->right = t;
                    return;
                }
            }
        }
    }

    node* deletefromBST(node* &root,int key){
           if(root==NULL){
               return root;
           }
           if(root->data==key){

               if(root->left==NULL && root->right==NULL){
                   delete root;
                   return NULL;

               }
               if(root->left!=NULL && root->right==NULL){
                   node* temp=root;
                   delete root;
                   return temp->left;
               }
               if(root->right!=NULL && root->left==NULL){
                   node* temp=root;
                   delete root;
                   return temp->right;
               }
               if(root->right!=NULL && root->left!=NULL){
                   int mini=Min(root->right);
                   root->data=mini;
                   root->right=deletefromBST(root->right,mini);
                   return root;
               }
           }
           else if(root->data > key){
               root->left=deletefromBST(root->left,key);
               return root;
           }
           else{
               root->right=deletefromBST(root->right,key);
               return root;
           }
       }


    void SwapBst(node * &curr){

        if(curr){
            swap(curr->left,curr->right);
            SwapBst(curr->left);
            SwapBst(curr->right);
        }
        else return;
    }
};
