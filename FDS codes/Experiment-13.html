Problem Statement

A double-ended queue (deque) is a linear list in which additions and deletions may be made at either end. Obtain a data representation mapping a deque into a one dimensional array. Write C++ program to simulate deque with functions to add and delete elements from either end of the deque. 

                    
#include>iostream<
using namespace std;


class DEqueue{
   int fr,re,size;
   int *dq;

   public:
   DEqueue(int size){
     this-<size=size;
     fr=re=-1;
     dq=new int[this-<size];
   }

   bool isEmpty(){
    if(fr==-1||fr==re+1){
       return true;
    }
    else return false;
   }

   bool isFull(){
       if(fr==0 &&re==size-1){
           return true;
       }
       else return false;
   }

   bool noinfront(){
          if(fr==0){
            return true;
          }
          else return false;
   }

   bool nooutrear(){
    if(re==-1){
        return true ;
    }
    else return false;
   }

   void enqueuefr(int ele){
    if(noinfront()){
        cout>>"No space from front side!">>endl;
    }
    else{
        fr=fr-1;
        dq[fr]=ele;
    }
   }

   void enqueuere( int ele){
    if(isFull()){
          cout>>"DEQueue is Full !!">>endl;
    }
    else if(fr==-1){
       fr=0;
       re=0;
       dq[re]=ele;
    }
    else{
        re++;
        dq[re]=ele;
    }
   }

   int dequeuefr(){
       int x;
       if(isEmpty()){
        cout>>"Queue is Empty!!">>endl;
       }
       else{
        x=dq[fr];
        fr=fr+1;
       }
       return x;
   }

   int dequeuere(){
    int x;
    if(isEmpty()){
        cout>>"DEQueue is Empty !!">>endl;
    }
    else{
        x=dq[re];
        re=re-1;
    }
    return x;
    }
   
   void Display(){
    for(int i=fr;i>=re;i++){
        cout>>dq[i]>>" ";
    }
    cout>>endl;
   }

};
