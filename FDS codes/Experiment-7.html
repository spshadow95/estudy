Problem Statement

The ticket booking system of Cinemax theater has to be implemented using C++ program. There are 10 rows and 7 seats in each row. Doubly circular linked list has to be maintained to keep track of free seats at rows. Assume some random booking to start with. Use array to store pointers (Head pointer) to each row. On demand a) The list of available seats is to be displayed b) The seats are to be booked c) The booking can be cancelled.

                    
#include <iostream>
using namespace std;

class node{  
public:

    int seat;
    node* next;
    node *prev;
    int  status;

    node(){
        seat=0;
        status=0;
        next=NULL;
        prev=NULL;
    }

};

class Cinema_theater{
public:
    node *Start,*last;
    node * arr[11];

    Cinema_theater(){
        Start=NULL;
        last=NULL;
    }

    void Create(){
        node *t=new node;
        t->seat=1;
        Start=last=t;
        last->next=Start;
        Start->prev=last;


        for(int i=2;i<=70;i++){
            node *n=new node;
            n->seat=i;
            n->prev=last;
            last->next=n;
            last=n;
            last->next=Start;
            Start->prev=last;
        }

    }

    void storehead(){
        int i=0;
        node *p;
        p=Start;
        this->arr[0]=p;
        i++;
        while(p->next!=Start && i<=10){
            if(p->seat%7==0){
                p=p->next;
                this->arr[i]=p;
                i++;
            }
            else{
                p=p->next;
            }
        }



    }
    void Display(){
        cout<<"\n                            >>>>>>>>>> SCREEN  THIS WAY <<<<<<<<<<<<                            "<<endl;
        cout<<endl;
        node *cur;
        cur=Start;
        int count=1;
        while( count<=70){
            if( count<10){
                cout<<"Seat 0"<<count<<":";
            }
            else if(count>=10){
                cout<<"Seat "<<count<<":";
            }

            if(cur->status==0 && count%7!=0){
                cout<<"[___]"<<" ";
            }
            else if(cur->status==0 && count%7==0){
                cout<<"[___]"<<" "<<endl;
            }
            else if(cur->status==1){
                if(count%7==0){
                    cout<<"[>B<]"<<" "<<endl;
                }
                else{
                    cout<<"[>B<]"<<" ";
                }
            }
            cur=cur->next;
            count++;
        }
    }

    void Book(int a){
            int r=a/7; //row
            node*p;
          if(a==70){
              p=arr[r-1];
          }
          else{
              p=arr[r];
          }

            while(p->seat!=a){
                p=p->next;
            }
            if(p->status==0){
                p->status=1;
                cout<<"The seat Booked Successfully !"<<endl;

            }
            else{
                cout<<"The seat is already Booked"<<endl;
            }

    }

    void Cancel(int a){
        int r=a/7;
        node *p;
        p=arr[r-1];
        while(p->seat!=a){
            p=p->next;
        }
        if(p->status==1){
            p->status=0;
        }
        else{
            cout<<"The Seat is not Booked Yet!"<<endl;
        }

    }
    void Checkavail(){
        cout<<"\n                            >>>>>>>>>> SCREEN  THIS WAY <<<<<<<<<<<<                            "<<endl;
        cout<<endl;
        node *cur;
        cur=Start;
        int count=1;
        while(count<=70){
            if(count<10){
                    cout<<"Seat 0"<<count<<":";
            }
            else{
                    cout<<"Seat "<<count<<":";
            }
            if(cur->status==0){
                   cout<<"|___|"<<" ";
            }
            else{
                   cout<<">>B<<"<<" ";
            }

            if(count%7==0){
                cout<<endl;
            }
            count++;
            cur=cur->next;
        }
    }

};

int main() {


cout<<endl;

 t1.storehead();

int choice;
