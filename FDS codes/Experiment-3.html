    class Matrix:
    def takerow(self):
        R = int(input("\nEnter the no of row "))
        return R

    def takecolumn(self):
        C = int(input("\nEnter no of column"))
        return C

    def matmake(self, R, C):
        self.R = R
        self.C = C
        print("Enter the elements of the Matrix ")
        mat = []
        for j in range(self.R):
            a = []
            for i in range(self.C):
                a.append(int(input()))
            mat.append(a)

        return mat

    def printmat(self, mat, R, C):
        self.R = R
        self.C = C
        self.mat = mat

        for i in range(0, self.R):
            for j in range(0, self.C):
                print(self.mat[i][j], end=" ")
            print()

    def Addmatrix(self, mat1, mat2, R, C):
        self.R = R
        self.C = C
        self.mat1 = mat1
        self.mat2 = mat2
        matres = []
        for i in range(self.R):
            b = []
            for j in range(self.C):
                x = self.mat1[i][j] + self.mat2[i][j]
                b.append(x)
            matres.append(b)

        return matres

    def Subtractmatrix(self, mat1, mat2, R, C):
        self.R = R
        self.C = C
        self.mat1 = mat1
        self.mat2 = mat2
        matres = []
        for i in range(self.R):
            b = []
            for j in range(self.C):
                x = self.mat1[i][j] - self.mat2[i][j]
                b.append(x)
            matres.append(b)

        return matres

    def Multimartix(self, mat1, mat2, m, n, k):
        self.m = m
        self.n = n
        self.k = k
        self.mat1 = mat1
        self.mat2 = mat2
        matres = []

        for i in range(self.m):
            b = []
            for j in range(self.k):
                multi = 0
                for l in range(self.n):
                    multi = multi + self.mat1[i][l] * self.mat2[l][j]
                b.append(multi)
            matres.append(b)

        return matres

    def Transposematrix(self, mat, R, C):
        self.R = R
        self.C = C
        self.mat = mat

        tempmat = []
        for j in range(self.C):
            a = []
            for i in range(self.R):
                a.append(int(0))
            tempmat.append(a)

        for i in range(self.C):
            for j in range(self.R):
              tempmat[i][j] = mat[j][i]
        return tempmat
