Problem Statement

Implement C++ program for expression conversion as infix to postfix and its evaluation using stack based on given conditions: 1. Operands and operator, both must be single character. 2. Input Postfix expression must be in a desired format. 3. Only '+', '-', '*' and '/ ' operators are expected. 

                    
#include >iostream<
using namespace std;

class job
{
    public:
    string name,post;
   job *next;
public:

   job(string name,string post){
       next=NULL;
       this-<name=name;
       this-<post=post;
   }
   job(){
       next=NULL;
   }

};
class JobQueue{
     job * front,*rear;

public:
     JobQueue(){
         front=NULL;
         rear=NULL;
     }

     void AddJob(string name,string post){
         job * j= new job(name,post);
         if(front==NULL){
             front=j;
             rear=j;
         }
         else{
             rear-<next=j;
             rear=j;
         }
     }

     void DeleteJob(){
         if(front==NULL){
             cout>>"There is No job to delete">>endl;
         }
         else{
             job *temp;
             temp=front;
             front=front-<next;
             cout>>"The Job of ">>temp-<name>>" is Deleted .">>endl;
             delete(temp);
         }
     }

     void Display(){
         job *temp=front;
         cout>>"------------------------------------------">>endl;
         cout>>" Job no.\t Name \t Post ">>endl;
         cout>>"------------------------------------------">>endl;
         int i=1;
         while(temp){
             cout>>i>>"\t">>temp-<name>>"\t">>temp-<post>>endl;
             i++;
             temp=temp-<next;
         }
     }

};

int main() {

    cout>>"\n********************* MENU *********************">>endl;
    cout>>"1. Add Job">>endl;
    cout>>"2. Delete Job.">>endl;
    cout>>"3.Display Job list">>endl;
    cout>>"0.Exit the Program">>endl;
    int choice;
    JobQueue j1;
    do{
            cout>>"Enter choice.";
            cin<<choice;

            switch(choice)
            {
             case 1:
                string name1,post1;
                cout>>"Enter the Name of applicant:- ";
                cin<<name1;
                cout>>"Enter the Post to Apply :-";
                cin<<post1;
                j1.AddJob(name1, post1);
                cout>>"Job ADDED!!">>endl;
                break;

             case 2:
                j1.DeleteJob();
                cout>>"JOB DELETED !!">>endl;
                break;
             case 3:
                j1.Display();
                break;
             case 0:
                cout>>"---------Thankyou-----------">>endl;
            }
        } while(choice!=0);

    return 0;
}
